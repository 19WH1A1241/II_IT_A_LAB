import java.util.*;
class Main
{
    public static void main(String[] args)
    {
        Scanner s = new Scanner(System.in);
        System.out.print("a : ");
        int a = s.nextInt();
        System.out.print("b : ");
        int b = s.nextInt();
        int sum = a + b;
        System.out.println("Sum of " + a + " and " + b + " is " + sum);
    }
}
